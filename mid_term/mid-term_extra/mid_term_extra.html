F0=sigma_m/a0
F0_vbi=sigma_m_vbi/a0
F1=F0-sigma/a0

def Ec(x):
    if(x<=L):
        return phi+F0*x
    if(x<=W):
        return F1*(x-L)+Ec(L)
    if(x<=W+xd):
        return Ec(W)+(F1+0.5*Nd*(x-W)/a0)*(x-W)
    return phi-Vbi

x=sp.linspace(0.1,2*W*1e7,1000) # x in nm
Ecb = lambda x : Ec(x*1e-7) # multiply by 1e-7 to convert x in nm to cm
E_cb = sp.vectorize(Ecb)(x)
E_vb = E_cb-Eg
